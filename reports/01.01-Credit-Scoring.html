def lemmatize_text(text, min_len=1, pymorph=None, global_counter=None, words_count=3, synonyms=None):
    """
        text    pymorph (pymorphy2.MorphAnalyzer).
         .     .
     pymorph      pymorphy2.MorphAnalyzer.
    stemmer       Pandas      
     .
       global_counter (collections.Counter),        
    words_count,          .

            text,  .
        min_len .
    """
    #  ,    
    if pymorph is None:
        pymorph = pymorphy2.MorphAnalyzer()

    #         
    words = [pymorph.parse(w)[0].normal_form for w in re.findall(r'[-]+', text.lower())]
        
    #        min_len
    words = [w for w in words if len(w) >= min_len]

    #    
    if synonyms is not None:
        words = [synonyms[w] if w in synonyms else w for w in words]
        words = [w for w in words if w is not None]
        
        
    #   
    if global_counter is not None:
        #      words_count ( ,     )
        word_tuples = [tuple(words[i : i + words_count]) for i in range(max(1, len(words) - words_count + 1))]
        #     
        global_counter += Counter(word_tuples)

    #     
    return ' '.join(words)
